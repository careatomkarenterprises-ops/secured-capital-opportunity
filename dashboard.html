FETCH DATA FROM GOOGLE SHEETS async function loadDashboardData(id) { showLoading(true); try { const response = await fetch(`${GOOGLE_SCRIPT_URL}?action=getPartner&accessKey=${id}`); const result = await response.json(); if (result.success) { const data = result.data; // Update UI with Sheet Data document.getElementById('investorName').textContent = data.FullName || "Investor"; document.getElementById('principalAmount').textContent = '₹' + Number(data.PrincipalAmount || 0).toLocaleString('en-IN'); document.getElementById('monthlyReturn').textContent = '₹' + Number(data.FixedMonthlyReturn || 0).toLocaleString('en-IN'); document.getElementById('totalReturns').textContent = '₹' + Number(data.TotalReturnsPaid || 0).toLocaleString('en-IN'); // Handle Days to Expiry (Simple Calculation) if(data.AgreementExpiryDate) { const expiry = new Date(data.AgreementExpiryDate); const diff = Math.ceil((expiry - new Date()) / (1000 * 60 * 60 * 24)); document.getElementById('daysToExpiry').textContent = diff > 0 ? diff : "Expired"; } // Load Transactions & KYC loadTransactions(id); loadKYCStatus(id); } } catch (e) { console.error("Dashboard Sync Failed:", e); } finally { showLoading(false); } } // 4. LOAD TRANSACTIONS TABLE async function loadTransactions(id) { const res = await fetch(`${GOOGLE_SCRIPT_URL}?action=getTransactions&accessKey=${id}`); const data = await res.json(); const tableBody = document.getElementById('recentTransactions'); tableBody.innerHTML = ''; if (data.success && data.transactions.length > 0) { data.transactions.forEach(tx => { const row = ` ${tx.TransactionDate || '-'} ${tx.TransactionType || 'Monthly Return'} ₹${tx.Amount || 0} ${tx.Status} ${tx.ReferenceNo || '-'} `; tableBody.insertAdjacentHTML('beforeend', row); }); } else { tableBody.innerHTML = 'No transactions found.'; } } // 5. BUTTON ACTIONS function logout() { localStorage.clear(); window.location.href = 'partner-portal.html'; } function showLoading(show) { document.querySelector('.loading-overlay').style.display = show ? 'flex' : 'none'; } // Add these to your buttons in the HTML document.querySelector('.btn-logout').onclick = logout;
